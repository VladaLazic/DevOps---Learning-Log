```
vi config.txt      # Open file
i                  # Insert mode
# Type the two lines
Esc                # Normal mode
:wq                # Save and quit

vi config.txt      # Open again
dd                 # Delete first line
o                  # New line below
timeout=30         # Type it
Esc                # Normal mode
:wq                # Save and quit
```
